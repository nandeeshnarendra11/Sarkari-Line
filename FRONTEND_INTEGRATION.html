A. Registration with OTP

const API_BASE = 'http://localhost:5000/api';

// Step 1: Register user
async function registerUser(userData) {
  try {
    const response = await fetch(`${API_BASE}/auth/register`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        firstName: userData.firstName,
        lastName: userData.lastName,
        email: userData.email,
        phone: userData.phone, // 10 digits only
        password: userData.password,
        confirmPassword: userData.confirmPassword,
        state: userData.state,
        district: userData.district,
        taluk: userData.taluk
      })
    });
    
    const data = await response.json();
    if (data.success) {
      console.log('Registration successful. OTP sent to phone.');
      // Show OTP verification form
      showOTPVerification(data.data.email);
    } else {
      alert('Error: ' + data.message);
    }
  } catch (error) {
    console.error('Registration failed:', error);
  }
}

// Step 2: Verify OTP
async function verifyOTP(email, otp) {
  try {
    const response = await fetch(`${API_BASE}/auth/verify-otp`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ email, otp })
    });
    
    const data = await response.json();
    if (data.success) {
      // Save token to localStorage
      localStorage.setItem('token', data.data.token);
      localStorage.setItem('user', JSON.stringify(data.data.user));
      console.log('Verified! Redirecting...');
      window.location.href = '/dashboard';
    } else {
      alert('Invalid OTP: ' + data.message);
    }
  } catch (error) {
    console.error('OTP verification failed:', error);
  }
}

C. Generate Queue Token

async function generateQueueToken(officeId, serviceType) {
  const token = localStorage.getItem('token');
  
  try {
    const response = await fetch(`${API_BASE}/queue/generate-token`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${token}`
      },
      body: JSON.stringify({
        officeId: officeId,
        serviceType: serviceType,
        priority: 'normal'
      })
    });
    
    const data = await response.json();
    if (data.success) {
      alert(`Your token number: ${data.data.token.tokenNumber}`);
      alert(`Position in queue: ${data.data.token.position}`);
      alert(`Est. wait time: ${data.data.token.estimatedWaitTime} mins`);
      // SMS will be sent automatically to user's registered phone
    } else {
      alert('Error: ' + data.message);
    }
  } catch (error) {
    console.error('Token generation failed:', error);
  }
}

F. Get All Offices

async function getOffices(filters = {}) {
  let url = `${API_BASE}/offices`;
  
  if (filters.state) url += `?state=${filters.state}`;
  if (filters.district) url += `&district=${filters.district}`;
  if (filters.serviceType) url += `&serviceType=${filters.serviceType}`;
  
  try {
    const response = await fetch(url, { method: 'GET' });
    const data = await response.json();
    
    if (data.success) {
      console.log(`Found ${data.count} offices`);
      displayOffices(data.data);
    }
  } catch (error) {
    console.error('Failed to fetch offices:', error);
  }
}

// Usage
getOffices({ state: 'Karnataka', district: 'Chikkaballapura' });

G. Helper Function - Set Auth Header

// Create a helper function for authenticated requests
async function apiCall(endpoint, method = 'GET', body = null) {
  const token = localStorage.getItem('token');
  const headers = {
    'Content-Type': 'application/json',
  };
  
  if (token) {
    headers['Authorization'] = `Bearer ${token}`;
  }
  
  const options = {
    method,
    headers,
  };
  
  if (body) {
    options.body = JSON.stringify(body);
  }
  
  try {
    const response = await fetch(`${API_BASE}${endpoint}`, options);
    const data = await response.json();
    return data;
  } catch (error) {
    console.error(`API call failed: ${endpoint}`, error);
    throw error;
  }
}

// Usage:
const tokens = await apiCall('/queue/my-tokens');
const user = await apiCall('/auth/me');
const newToken = await apiCall('/queue/generate-token', 'POST', {
  officeId: 'xxx',
  serviceType: 'Passport'
});

📱 SMS Notifications

The system automatically sends SMS to user's registered phone number:

  • Registration OTP: 6-digit code for account verification
  • Registration Confirmation: After successful OTP verification
  • Queue Status: When your token is called at office
  • Appointment Reminder: Before scheduled appointment

🔐 Authentication Flow

1. Register

POST /api/auth/register
Returns: User ID and message to verify OTP

2. Verify OTP

POST /api/auth/verify-otp
Returns: JWT Token (save to localStorage)

3. Login

POST /api/auth/login
Returns: JWT Token (save to localStorage)

4. Use Token

Add to every authenticated request:
Authorization: Bearer {token}

📊 API Endpoints Summary

Method Endpoint Description Auth Required
POST /api/auth/register Register new user No
POST /api/auth/verify-otp Verify OTP No
POST /api/auth/login Login user No
GET /api/auth/me Get current user Yes
POST /api/queue/generate-token Generate queue token Yes
GET /api/queue/my-tokens Get user's tokens Yes
GET /api/queue/office/:id/status Get office queue status No
GET /api/offices Get all offices No
GET /api/offices/search Search offices No
GET /api/offices/:id Get office details No

🔑 Key Points

Phone Number Format: Use 10 digits only (e.g., 9876543210). The backend will add +91 automatically for SMS.
Token Storage: Save JWT token in localStorage to persist user session. Clear on logout.
Error Handling: Always check response.success and handle error messages appropriately.
CORS: Backend allows requests from FRONTEND_URL (default: http://localhost:3000). Update in .env if needed.
